CREATE TRIGGER trg_min_salary_update
BEFORE UPDATE ON Employee
FOR EACH ROW
BEGIN
    IF NEW.salary < 10000 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'ERROR: Updated salary cannot be less than 10,000!';
    END IF;
END$$

-- Step 4: Stored Procedure using CURSOR
CREATE PROCEDURE sp_display_employees()
BEGIN
    DECLARE v_id    INT;
    DECLARE v_name  VARCHAR(100);
    DECLARE v_dept  VARCHAR(50);
    DECLARE v_sal   DECIMAL(10,2);
    DECLARE v_total DECIMAL(10,2) DEFAULT 0;
    DECLARE done    INT DEFAULT 0;

    DECLARE emp_cursor CURSOR FOR
        SELECT emp_id, emp_name, department, salary FROM Employee;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

    OPEN emp_cursor;

    read_loop: LOOP
        FETCH emp_cursor INTO v_id, v_name, v_dept, v_sal;
        IF done = 1 THEN
            LEAVE read_loop;
        END IF;
        SELECT v_id AS EmpID, v_name AS Name, v_dept AS Department, v_sal AS Salary;
        SET v_total = v_total + v_sal;
    END LOOP;

    CLOSE emp_cursor;
    SELECT CONCAT('TOTAL SALARY OF ALL EMPLOYEES: Rs. ', v_total) AS Summary;
END$$

DELIMITER ;

-- Step 5: Call the Procedure

-- Query 1: Employees earning more than average salary
SELECT emp_id, emp_name, salary
FROM Employees
WHERE salary > (SELECT AVG(salary) FROM Employees);

-- Query 2: Employee(s) with the highest salary
SELECT emp_id, emp_name, salary
FROM Employees
WHERE salary = (SELECT MAX(salary) FROM Employees);

-- Query 3: Salary greater than dept average (correlated subquery)
SELECT e.emp_id, e.emp_name, e.salary, e.dept_id
FROM Employees e
WHERE e.salary > (
    SELECT AVG(e2.salary)
    FROM Employees e2
    WHERE e2.dept_id = e.dept_id
);

-- Query 4: Employees in depts where MAX salary > 60,000
SELECT e.emp_id, e.emp_name, e.salary, d.dept_name
FROM Employees e
JOIN Departments d ON e.dept_id = d.dept_id
WHERE e.dept_id IN (
    SELECT dept_id FROM Employees
    GROUP BY dept_id
    HAVING MAX(salary) > 60000
);

-- Query 5: Employees NOT assigned to any department
SELECT emp_id, emp_name
FROM Employees
WHERE dept_id NOT IN (
    SELECT dept_id FROM Departments
) OR dept_id IS NULL;

-- Query 6: EXISTS - employees belonging to existing departments
SELECT e.emp_id, e.emp_name
FROM Employees e
WHERE EXISTS (
    SELECT 1 FROM Departments d
    WHERE d.dept_id = e.dept_id
);

CREATE DATABASE IF NOT EXISTS grade_db;
USE grade_db;

CREATE TABLE Stud_Marks (
    id          INT          PRIMARY KEY AUTO_INCREMENT,
    name        VARCHAR(100) NOT NULL,
    total_marks INT          NOT NULL
);

CREATE TABLE Result (
    roll  INT          PRIMARY KEY AUTO_INCREMENT,
    name  VARCHAR(100),
    class VARCHAR(30)
);

INSERT INTO Stud_Marks (name, total_marks) VALUES
('Alice',   1200),
('Bob',     950),
('Carol',   870),
('David',   800),
('Eva',     1450),
('Frank',   910),
('Grace',   835),
('Hina',    780),
('Ivan',    1100),
('Jasmine', 990);

DELIMITER $$

CREATE PROCEDURE proc_Grade()
BEGIN
    DECLARE v_name  VARCHAR(100);
    DECLARE v_marks INT;
    DECLARE v_class VARCHAR(30);
    DECLARE done    INT DEFAULT 0;

    DECLARE stud_cursor CURSOR FOR
        SELECT name, total_marks FROM Stud_Marks;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

    TRUNCATE TABLE Result;

    OPEN stud_cursor;

    grade_loop: LOOP
        FETCH stud_cursor INTO v_name, v_marks;
        IF done = 1 THEN
            LEAVE grade_loop;
        END IF;

        IF v_marks >= 990 AND v_marks <= 1500 THEN
            SET v_class = 'Distinction';
        ELSEIF v_marks >= 900 AND v_marks <= 989 THEN
            SET v_class = 'First Class';
        ELSEIF v_marks >= 825 AND v_marks <= 899 THEN
            SET v_class = 'Higher Second Class';
        ELSE
            SET v_class = 'Pass / Fail';
        END IF;

        INSERT INTO Result (name, class) VALUES (v_name, v_class);
    END LOOP;

    CLOSE stud_cursor;

    SELECT * FROM Result;
    SELECT CONCAT('Total students processed: ', COUNT(*)) AS Summary FROM Result;
END$$

DELIMITER ;

-- Call the Procedure
CALL proc_Grade();

-- Verify Results
SELECT R.roll, SM.name, SM.total_marks, R.class
FROM Result R
JOIN Stud_Marks SM ON R.name = SM.name
ORDER BY SM.total_marks DESC;

  { name: "Bob",   product: "Trousers", category: "Clothing",    amt: 2800,  date: new Date("2024-02-05") }
]);

// Method 1: Aggregation Pipeline
db.customer_total.drop();

db.orders.aggregate([
  { $group: { _id: "$name", totalSales: { $sum: "$amt" }, orderCount: { $count: {} }, avgOrderAmt: { $avg: "$amt" } } },
  { $project: { _id: 0, customer: "$_id", totalSales: 1, orderCount: 1, avgOrderAmt: { $round: ["$avgOrderAmt", 2] } } },
  { $out: "customer_total" }
]);

db.customer_total.find().sort({ totalSales: -1 }).pretty();

// Method 2: JavaScript forEach (simulates cursor iteration)
db.customer_total.drop();

let customerMap = {};

db.orders.find().forEach(function(order) {
  let cust = order.name;
  if (!customerMap[cust]) {
    customerMap[cust] = { totalSales: 0, orderCount: 0 };
  }
  customerMap[cust].totalSales += order.amt;
  customerMap[cust].orderCount += 1;
});

for (let cust in customerMap) {
  db.customer_total.insertOne({
    customer:   cust,
    totalSales: customerMap[cust].totalSales,
    orderCount: customerMap[cust].orderCount,
    avgAmt:     parseFloat((customerMap[cust].totalSales / customerMap[cust].orderCount).toFixed(2))
  });
}

db.customer_total.find().sort({ totalSales: -1 }).pretty();

DELIMITER $$

CREATE PROCEDURE transfer_amount(
    IN from_acc INT,
    IN to_acc   INT,
    IN amount   DECIMAL(12,2)
)
BEGIN
    DECLARE sender_bal DECIMAL(12,2);

    SELECT balance INTO sender_bal FROM Accounts WHERE acc_id = from_acc FOR UPDATE;

    START TRANSACTION;

    IF sender_bal < amount THEN
        ROLLBACK;
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'ERROR: Insufficient balance. Transfer cancelled.';
    ELSE
        UPDATE Accounts SET balance = balance - amount WHERE acc_id = from_acc;
        SAVEPOINT after_deduction;
        UPDATE Accounts SET balance = balance + amount WHERE acc_id = to_acc;
        COMMIT;
        SELECT CONCAT('Transfer of Rs.', amount, ' SUCCESSFUL!') AS Status;
    END IF;
END$$

DELIMITER ;

CREATE DATABASE IF NOT EXISTS trigger_db;
USE trigger_db;

CREATE TABLE Products (
    p_id         INT           PRIMARY KEY AUTO_INCREMENT,
    pname        VARCHAR(100),
    price        DECIMAL(10,2),
    stock        INT DEFAULT 0,
    last_updated DATETIME
);

CREATE TABLE Audit_Log (
    log_id     INT         PRIMARY KEY AUTO_INCREMENT,
    table_name VARCHAR(50),
    action     VARCHAR(20),
    old_value  TEXT,
    new_value  TEXT,
    changed_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

INSERT INTO Products (pname, price, stock) VALUES
('Laptop',55000,10), ('Phone',25000,20), ('Tablet',18000,15);

DELIMITER $$

-- BEFORE INSERT Trigger: Validates price > 0
CREATE TRIGGER trg_before_product_insert
BEFORE INSERT ON Products
FOR EACH ROW
BEGIN
    IF NEW.price <= 0 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Price must be greater than 0';
    END IF;
    SET NEW.last_updated = NOW();
END$$

-- AFTER INSERT Trigger: Logs every new product insertion
CREATE TRIGGER trg_after_product_insert
AFTER INSERT ON Products
FOR EACH ROW
BEGIN
    INSERT INTO Audit_Log (table_name, action, old_value, new_value)
    VALUES ('Products', 'INSERT', NULL,
            CONCAT('ID=', NEW.p_id, ', Name=', NEW.pname, ', Price=', NEW.price, ', Stock=', NEW.stock));
END$$

-- BEFORE UPDATE Trigger: Prevents price reduction > 50%
CREATE TRIGGER trg_before_product_update
BEFORE UPDATE ON Products
FOR EACH ROW
BEGIN
    IF NEW.price < (OLD.price * 0.50) THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Price cannot be reduced by more than 50%';
    END IF;
    SET NEW.last_updated = NOW();
END$$

-- AFTER UPDATE Trigger: Logs every product update
CREATE TRIGGER trg_after_product_update
AFTER UPDATE ON Products
FOR EACH ROW
BEGIN
    INSERT INTO Audit_Log (table_name, action, old_value, new_value)
    VALUES ('Products', 'UPDATE',
            CONCAT('Price=', OLD.price, ', Stock=', OLD.stock),
            CONCAT('Price=', NEW.price, ', Stock=', NEW.stock));
END$$

-- BEFORE DELETE Trigger: Prevents deletion if stock > 15
CREATE TRIGGER trg_before_product_delete
BEFORE DELETE ON Products
FOR EACH ROW
BEGIN
    IF OLD.stock > 15 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cannot delete product with stock > 15';
    END IF;
END$$

-- AFTER DELETE Trigger: Logs deleted records
CREATE TRIGGER trg_after_product_delete
AFTER DELETE ON Products
FOR EACH ROW
BEGIN
    INSERT INTO Audit_Log (table_name, action, old_value, new_value)
    VALUES ('Products', 'DELETE',
            CONCAT('ID=', OLD.p_id, ', Name=', OLD.pname, ', Price=', OLD.price), NULL);
END$$

DELIMITER ;
